// 异步循环打印
function sleep(time, val) {
  return new Promise(function(resolve, reject) {
    setTimeout(() => {
      resolve(val)
    }, time)
  })
}

var run = async function() {
  for (var i = 0; i < 10; i++) {
    var result = await sleep(1000, i)
    console.log(result)
  }
}
